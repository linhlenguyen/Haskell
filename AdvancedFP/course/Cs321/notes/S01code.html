fun lastone x = hd (rev x)
fun prefix x = rev (tl (rev x))

fun mymap f [] = []
  | mymap f (x::xs) = (f x):: (mymap f xs);

fun filter p [] = []
  | filter p (x::xs) =
     if (p x) then x :: (filter p xs) else (filter p xs);
