module CreatingListsExample where

x1 = [2,3,45]

x2 = 99 : x1

(y : ys) = [6,78,2-6,8]  -- 6 : 78 : (2-6) : 8 : []

(a,b) = (45,"time")

(q,(m:ms)) = (789,x2)

(w : (x : z)) = [1,2,3,4,5]

ex3 = 1 : (2 : (7 : (6 : (4 : []))))   -- [1,2,7,6,4]

ex4 = 1 : 2 : 7 : 6 : 4 : []

a1 = []
a2 = 3 : a1
a3 = 6 : a2
